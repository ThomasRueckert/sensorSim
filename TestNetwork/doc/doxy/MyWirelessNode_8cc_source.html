93 void MyWirelessNode::handleMessage(cMessage *msg)
94 {
95  std::string msgName = msg->getName();
96  if (msg->isSelfMessage()) {
97  ev.bubble(this, "Selfmessage received");
98  } else if (msg->getSenderModuleId() == 16){
99  EV << "MyWirelessNode: Message from Network received" << endl;
100  if (msgName == "GET type") {
101  this->handleGetType(msg);
102  }
103  } else {
104  if (msgName.substr(0,4) == "POST") {
105 
106  SimpleSensorData* data = (SimpleSensorData*) msg->getParList().remove("data");
107  this->sensorData = data->sensorData;
108  int** arr = new int*[400];
109  for (int i = 0; i < 400; i++) {
110  arr[i] = new int[550];
111  for (int j = 0; j <550; j++) {
112  arr[i][j] = data->sensorData[i][j];
113  }
114  }
115  this->sensorData = arr;
116  delete data;
117 
118  } else {
119  ExtendedMessage *extmsg = check_and_cast<ExtendedMessage *>(msg);
120  int hopcount = extmsg->getHopCount();
121  hopCountVector.record(hopcount);
122  hopCountStats.collect(hopcount);
123  numReceived++;
124  ev.bubble(this, msg->getName());
125  if (!msg->isSelfMessage()) {
126  this->requestData();
127  }
128  if (ev.isGUI()) {
129  updateDisplay();
130  }
131  }
132  }
133  delete msg;
134 }
