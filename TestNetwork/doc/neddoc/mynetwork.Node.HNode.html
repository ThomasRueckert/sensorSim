module HNode extends MyWirelessNode

    gates:
        inout toSensor[1];
        inout worldGate[1];

    submodules:
        HumiditySensor: HumiditySensor {
            @display("p=140,310");
        }

    connections:
        toSensor[0] <--> {  delay = 0ms; } <--> HumiditySensor.toNode;
        worldGate[0] <--> {  delay = measureTime; } <--> HumiditySensor.worldDataGate;
